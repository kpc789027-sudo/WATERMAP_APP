// ============================================= // 산림 현장조사 전자야장 - Apps Script (V20) // 수정: FOLDER_ID 동적 수신, 주차 열 추가 // ============================================= const SOIL_SHEET = '현장조사(입지 및 토양)'; const TREE_SHEET = '현장조사(임목)'; const PHOTO_SHEET = '사진목록'; const PLOT_SHEET = '조사팀관리'; // FOLDER_ID는 HTML 설정탭에서 payload.folderId로 전달받아 사용합니다 // ============================================= // 헤더 정의 // ============================================= const SOIL_HEADER = [ '유역코드','중유역명','조사지번호','조사일', '날씨','수행사','조사책임자','조사지이동필요성', 'X좌표(EPSG5186)','Y좌표(EPSG5186)', '침식상태','퇴적양식','풍노출도','암석노출도', '유기물층L','유기물층F','유기물층H', 'A층토심','B층토심', 'A층토색','B층토색', '풍화정도','A층건습도','B층건습도', 'A층자갈함량','B층자갈함량', 'A층견밀도','B층견밀도', '토양형', '100cc_10cm','100cc_30cm', '이화학성_10cm','이화학성_30cm', '특이사항','주차' ]; const TREE_HEADER = [ '유역명','유역코드','조사지번호','수행사','임종', '수번','수종명','흉고직경(cm)','수고(m)','주차' ]; const PHOTO_HEADER = [ '조사지번호','팀코드','파일명','Drive URL','저장일시' ]; // ============================================= // 헤더 자동 생성 // ============================================= function ensureHeader(sheet, header) { if (sheet.getLastRow() === 0) { sheet.appendRow(header); sheet.getRange(1, 1, 1, header.length) .setBackground('#0A5C45') .setFontColor('white') .setFontWeight('bold'); sheet.setFrozenRows(1); } } // ============================================= // 메인 진입점 // ============================================= function doPost(e) { // CORS 허용 const output = (obj) => ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); try { // text/plain 또는 application/json 둘 다 처리 const raw = e.postData ? e.postData.contents : ''; let payload; try { payload = JSON.parse(raw); } catch (_) { try { payload = JSON.parse(e.parameter.data || '{}'); } catch (__) { return output({ status: 'error', message: 'parse failed' }); } } const ss = SpreadsheetApp.getActiveSpreadsheet(); let result = { status: 'ok' }; if (payload.type === 'soil') { result = handleSoil(ss, payload.row); } else if (payload.type === 'tree') { result = handleTree(ss, payload.rows, payload.overwrite !== false); } else if (payload.type === 'photo') { result = handlePhoto(ss, payload); } else if (payload.type === 'photoCheck') { result = handlePhotoCheck(ss, payload); } else { result = { status: 'error', message: 'unknown type: ' + payload.type }; } return output(result); } catch (err) { return output({ status: 'error', message: err.toString() }); } } // ============================================= // 입지토양 저장 // ============================================= function handleSoil(ss, row) { const sheet = ss.getSheetByName(SOIL_SHEET) || ss.insertSheet(SOIL_SHEET); ensureHeader(sheet, SOIL_HEADER); const plotNo = row[2]; const data = sheet.getDataRange().getValues(); let targetRow = -1; for (let i = 1; i < data.length; i++) { if (data[i][2] == plotNo) { targetRow = i + 1; break; } } if (targetRow > 0) { sheet.getRange(targetRow, 1, 1, row.length).setValues([row]); } else { sheet.appendRow(row); } return { status: 'ok', sheet: SOIL_SHEET }; } // ============================================= // 임목조사 저장 (조사지번호 기준 덮어쓰기) // ============================================= function handleTree(ss, rows, overwrite) { if (!rows || rows.length === 0) return { status: 'ok', message: 'no tree data' }; const sheet = ss.getSheetByName(TREE_SHEET) || ss.insertSheet(TREE_SHEET); ensureHeader(sheet, TREE_HEADER); const plotNo = rows[0][2]; const data = sheet.getDataRange().getValues(); // 기존 행 존재 여부 확인 const existingRows = []; for (let i = data.length - 1; i >= 1; i--) { if (data[i][2] == plotNo) existingRows.push(i + 1); } // overwrite 플래그가 false이고 기존 행이 있으면 중복 알림 반환 if (existingRows.length > 0 && overwrite === false) { return { status: 'duplicate', sheet: TREE_SHEET, plotNo: plotNo, count: existingRows.length }; } // 기존 행 삭제 후 새로 추가 existingRows.forEach(r => sheet.deleteRow(r)); rows.forEach(r => sheet.appendRow(r)); return { status: 'ok', sheet: TREE_SHEET, count: rows.length, overwritten: existingRows.length > 0 }; } // ============================================= // 사진 Drive 저장 (파일명 기준 덮어쓰기) // ============================================= function handlePhoto(ss, payload) { try { // folderId 검증 if (!payload.folderId) { return { status: 'error', message: 'Drive 폴더 ID가 없습니다. 설정탭에서 확인해주세요.' }; } if (!payload.base64 || payload.base64.length === 0) { return { status: 'error', message: '사진 데이터가 비어있습니다: ' + payload.filename }; } // 드라이브 폴더 생성/접근 let folder; try { folder = getDriveFolder(payload.folderId, payload.plotNo, payload.teamCode, payload.weekCode); } catch(folderErr) { return { status: 'error', message: '폴더 접근 실패: ' + folderErr.message }; } // blob 생성 const blob = Utilities.newBlob( Utilities.base64Decode(payload.base64), 'image/jpeg', payload.filename ); // 드라이브에 같은 파일명 있으면 덮어쓰기 const existing = folder.getFilesByName(payload.filename); if (existing.hasNext()) existing.next().setTrashed(true); // 파일 생성 — 실패 시 에러 반환 let file; try { file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); } catch(fileErr) { return { status: 'error', message: '드라이브 저장 실패: ' + fileErr.message }; } // 파일 URL 확인 const fileUrl = file.getUrl(); if (!fileUrl) { return { status: 'error', message: '파일 URL 획득 실패: ' + payload.filename }; } // 사진목록 시트 기록 try { const sheet = ss.getSheetByName(PHOTO_SHEET) || ss.insertSheet(PHOTO_SHEET); ensureHeader(sheet, PHOTO_HEADER); const sheetData = sheet.getDataRange().getValues(); let existingSheetRow = -1; for (let i = 1; i < sheetData.length; i++) { if (sheetData[i][2] == payload.filename) { existingSheetRow = i + 1; break; } } const newRow = [ payload.plotNo, payload.teamCode, payload.filename, fileUrl, new Date().toLocaleString('ko-KR') ]; if (existingSheetRow > 0) { sheet.getRange(existingSheetRow, 1, 1, newRow.length).setValues([newRow]); } else { sheet.appendRow(newRow); } } catch(sheetErr) { // 시트 기록 실패해도 드라이브엔 저장됐으므로 경고로 처리 return { status: 'warn', url: fileUrl, message: '드라이브 저장 성공, 시트 기록 실패: ' + sheetErr.message }; } return { status: 'ok', url: fileUrl }; } catch(e) { return { status: 'error', message: '사진 업로드 중 오류: ' + e.message }; } } // ============================================= // 사진 중복 확인 (업로드 전 체크용) // ============================================= function handlePhotoCheck(ss, payload) { const folder = getDriveFolder(payload.folderId, payload.plotNo, payload.teamCode, payload.weekCode); const existing = folder.getFilesByName(payload.filename); return { status: 'ok', exists: existing.hasNext(), filename: payload.filename }; } // ============================================= // Drive 폴더 구조: 루트 > 주차 > 팀코드 > 조사지번호 // ============================================= function getDriveFolder(folderId, plotNo, teamCode, weekCode) { if (!folderId) throw new Error('Drive 폴더 ID가 전달되지 않았습니다. HTML 설정탭에서 Google Drive 폴더 ID를 확인해주세요.'); const root = DriveApp.getFolderById(folderId); const weekFolder = getOrCreateFolder(root, weekCode || '주차미정'); const teamFolder = getOrCreateFolder(weekFolder, teamCode || '미지정팀'); return getOrCreateFolder(teamFolder, plotNo || '미지정조사지'); } function getOrCreateFolder(parent, name) { // 같은 이름 폴더 모두 조회 — 중복 폴더 있으면 빈 것 제거 후 첫 번째 사용 const iter = parent.getFoldersByName(name); let target = null; const extras = []; while (iter.hasNext()) { const f = iter.next(); if (target === null) { target = f; } else { extras.push(f); // 중복 폴더 수집 } } // 중복 폴더: 파일 없는 것만 휴지통으로 이동 extras.forEach(function(f) { try { if (f.getFiles().hasNext() === false) f.setTrashed(true); } catch(e) {} }); // 없으면 새로 생성 return target !== null ? target : parent.createFolder(name); } // ============================================= // GET 요청 (팀 목록, 포인트 불러오기) // ============================================= function doGet(e) { const action = e.parameter.action; const ss = SpreadsheetApp.getActiveSpreadsheet(); if (action === 'getTeams') { const sheet = ss.getSheetByName(PLOT_SHEET); if (!sheet) return json({ teams: [] }); const data = sheet.getDataRange().getValues(); const teams = [...new Set( data.slice(1).map(r => (r[6] || '').toString().trim()).filter(Boolean) )]; return json({ teams }); } // 현장조사(입지 및 토양) 시트에서 완료된 조사지번호만 빠르게 반환 if (action === 'getCompletedPlots') { const soilSh = ss.getSheetByName(SOIL_SHEET); const completed = []; if (soilSh && soilSh.getLastRow() > 1) { const rows = soilSh.getDataRange().getValues(); rows.slice(1).forEach(r => { const pn = (r[2] || '').toString().trim(); if (pn) completed.push(pn); }); } return json({ completed }); } if (action === 'getPlots') { const team = e.parameter.team || ''; const sheet = ss.getSheetByName(PLOT_SHEET); if (!sheet) return json({ plots: [] }); const data = sheet.getDataRange().getValues(); // 현장조사(입지 및 토양) 시트에서 업로드된 조사지번호 목록 추출 const soilSheetForCheck = ss.getSheetByName(SOIL_SHEET); const uploadedPlotNos = new Set(); if (soilSheetForCheck && soilSheetForCheck.getLastRow() > 1) { const soilData = soilSheetForCheck.getDataRange().getValues(); soilData.slice(1).forEach(r => { const pn = (r[2] || '').toString().trim(); if (pn) uploadedPlotNos.add(pn); }); } const plots = data.slice(1) .filter(r => r[6] == team && (r[8] || '').toString().trim() === '') .map(r => ({ basin: r[0], basinCode: r[1], plotNo: r[2].toString().trim(), x: r[3] ? r[3].toString().trim() : '', y: r[4] ? r[4].toString().trim() : '', address: r[5] ? r[5].toString().trim() : '', team: r[6], week: r[7], // 현장조사 시트에 있으면 완료 done: uploadedPlotNos.has(r[2].toString().trim()) ? '완료' : '' })); return json({ plots }); } // 구글시트에서 특정 조사지 데이터 불러오기 if (action === 'getSurveyData') { const plotNo = e.parameter.plotNo || ''; if (!plotNo) return json({ error: '조사지번호 필요' }); const result = {}; // 입지토양 데이터 const soilSheet = ss.getSheetByName(SOIL_SHEET); if (soilSheet) { const soilData = soilSheet.getDataRange().getValues(); for (let i = 1; i < soilData.length; i++) { if (soilData[i][2] == plotNo) { result.soilRow = soilData[i]; break; } } } // 임목조사 데이터 const treeSheet = ss.getSheetByName(TREE_SHEET); if (treeSheet) { const treeData = treeSheet.getDataRange().getValues(); result.treeRows = []; for (let i = 1; i < treeData.length; i++) { if (treeData[i][2] == plotNo) { result.treeRows.push(treeData[i]); } } } if (!result.soilRow && result.treeRows && result.treeRows.length === 0) { return json({ error: '데이터 없음', plotNo }); } return json({ status: 'ok', ...result }); } return json({ status: 'ok', message: '산림 전자야장 API' }); } function json(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
