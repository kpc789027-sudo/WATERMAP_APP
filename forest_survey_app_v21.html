Apps Script 코드

설정 순서
① Google Sheets에서 시트 탭 4개 생성:
   • 현장조사(입지 및 토양)
   • 현장조사(임목)
   • 사진목록
   • 조사팀관리 ← 포인트 활성화 관리 시트
② 조사팀관리 시트 헤더(1행): A유역명 / B유역코드 / C표준지번호 / D조사팀 / E주차 / F완료유무 / G완료일자 / H비고
   ※ E열(주차)에 값(예: 1주차)이 있는 행만 앱에 활성화됩니다
   ※ E열이 빈칸이면 해당 표준지는 드롭다운에 표시되지 않습니다
③ 상단 메뉴 → 확장 프로그램 → Apps Script
④ 아래 코드 전체 붙여넣기 → 저장
⑤ 배포 → 새 배포 → 웹앱 → 액세스: 모든 사용자 → URL 복사
⑥ 위 'Google 연동' 설정에 URL 붙여넣기
※ 헤더(항목명)는 첫 저장 시 자동으로 생성됩니다
// ============================================= // 시트 이름 상수 // ============================================= const SOIL_SHEET = '현장조사(입지 및 토양)'; const TREE_SHEET = '현장조사(임목)'; const PHOTO_SHEET = '사진목록'; const PLOT_SHEET = '조사팀관리'; // ← 포인트 활성화 관리 const FOLDER_ID = '1DjV_ieloSuuMYo9C47MB9B_Tw44UNb1O'; // ============================================= // 헤더 정의 // ============================================= const SOIL_HEADER = [ '유역명','유역코드','표준지번호','조사일','날씨', '좌표X(EPSG5186)','좌표Y(EPSG5186)', '침식상태','퇴적양식','풍노출도','암석노출도', '유기물층L','유기물층F','유기물층H', '토심A','토심B','토색A','토색B','풍화정도', '건습도A','건습도B','자갈함량A','자갈함량B', '견밀도A','견밀도B','토양형', '100cc_10cm','100cc_30cm','이화학성_10cm','이화학성_30cm', '특이사항' ]; const TREE_HEADER = [ '유역명','유역코드','표준지번호','수행사','임종', '번호','수종명','흉고직경(cm)','수고(m)' ]; const PHOTO_HEADER = ['표준지번호','파일명','저장일시','Drive링크']; const PLOT_HEADER = ['표준지번호','조사팀','주차','완료유무','완료일자','비고']; // ============================================= // 헤더 자동 생성 함수 // ============================================= function ensureHeader(sheet, header) { if (sheet.getLastRow() === 0) { sheet.appendRow(header); sheet.getRange(1, 1, 1, header.length) .setFontWeight('bold') .setBackground('#E1F5EE'); sheet.setFrozenRows(1); } } // ============================================= // doPost: 현장 데이터 수신 (토양/임목/사진) // ============================================= function doPost(e) { const raw = (e.parameter && e.parameter.data) ? e.parameter.data : (e.postData ? e.postData.contents : '{}'); const data = JSON.parse(raw); const ss = SpreadsheetApp.getActiveSpreadsheet(); if (data.type === 'soil') { const sh = ss.getSheetByName(SOIL_SHEET); ensureHeader(sh, SOIL_HEADER); // 중복 표준지번호 → 해당 행 덮어쓰기 const plotNo = data.row[2]; if (plotNo) { const existing = sh.getDataRange().getValues(); for (let i = 1; i < existing.length; i++) { if (String(existing[i][2]).trim() === String(plotNo).trim()) { sh.getRange(i + 1, 1, 1, data.row.length).setValues([data.row]); return jsonResponse({ status: 'ok', updated: true }); } } } sh.appendRow(data.row); } else if (data.type === 'tree') { const sh = ss.getSheetByName(TREE_SHEET); ensureHeader(sh, TREE_HEADER); data.rows.forEach(r => sh.appendRow(r)); } else if (data.type === 'photo') { const rootFolder = DriveApp.getFolderById(FOLDER_ID); // 1. 조사팀 폴더 (없으면 생성) const teamName = data.teamCode || '미지정팀'; let teamFolder; const teamIter = rootFolder.getFoldersByName(teamName); teamFolder = teamIter.hasNext() ? teamIter.next() : rootFolder.createFolder(teamName); // 2. 표준지번호 폴더 (없으면 생성) const plotNo = data.plotNo || data.filename.replace(/_.*/, ''); let plotFolder; const plotIter = teamFolder.getFoldersByName(plotNo); plotFolder = plotIter.hasNext() ? plotIter.next() : teamFolder.createFolder(plotNo); // 3. 사진 파일 저장 const blob = Utilities.newBlob( Utilities.base64Decode(data.base64), 'image/jpeg', data.filename ); const file = plotFolder.createFile(blob); // 4. 사진목록 시트에 기록 const sh = ss.getSheetByName(PHOTO_SHEET); ensureHeader(sh, PHOTO_HEADER); sh.appendRow([ plotNo, data.filename, new Date().toLocaleString('ko-KR'), file.getUrl() ]); } return jsonResponse({ status: 'ok' }); } // ============================================= // doGet: 포인트 목록 조회 (조사팀관리 시트 기반) // // [조사팀관리 시트 열 구조] // A열: 유역명 // B열: 유역코드 // C열: 표준지번호 // D열: 조사팀 // E열: 주차 (★ 활성화 핵심 - 값 있으면 활성 / 빈칸이면 비활성) // F열: 완료유무 // G열: 완료일자 // H열: 비고 // ============================================= function doGet(e) { const action = e && e.parameter && e.parameter.action; const ss = SpreadsheetApp.getActiveSpreadsheet(); if (action === 'getTeams') { const sh = ss.getSheetByName(PLOT_SHEET); if (!sh) return jsonResponse({ error: '조사팀관리 시트가 없습니다.' }); const rows = sh.getDataRange().getValues(); const teams = new Set(); for (let i = 1; i < rows.length; i++) { const team = String(rows[i][3] || '').trim(); // D열: 조사팀 if (team) teams.add(team); } return jsonResponse({ teams: [...teams] }); } if (action === 'getPlots') { const team = String(e.parameter.team || '').trim(); const sh = ss.getSheetByName(PLOT_SHEET); if (!sh) return jsonResponse({ error: '조사팀관리 시트가 없습니다. 시트 탭 이름을 확인하세요.' }); const rows = sh.getDataRange().getValues(); const plots = []; for (let i = 1; i < rows.length; i++) { const basin = String(rows[i][0] || '').trim(); // A열: 유역명 const basinCode= String(rows[i][1] || '').trim(); // B열: 유역코드 const plotNo = String(rows[i][2] || '').trim(); // C열: 표준지번호 const rowTeam = String(rows[i][3] || '').trim(); // D열: 조사팀 const week = String(rows[i][4] || '').trim(); // E열: 주차 (빈칸=비활성) const done = String(rows[i][5] || '').trim(); // F열: 완료유무 const doneDate = String(rows[i][6] || '').trim(); // G열: 완료일자 const note = String(rows[i][7] || '').trim(); // H열: 비고 // ★ 활성 조건: 표준지번호 있고 + 팀 일치 + E열(주차)에 값이 있어야 활성 if (plotNo && rowTeam === team && week !== '') { plots.push({ plotNo, basin, basinCode, week, done, doneDate, note }); } } return jsonResponse({ plots, team }); } return jsonResponse({ status: 'ok' }); } // ============================================= // JSON 응답 헬퍼 // ============================================= function jsonResponse(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
